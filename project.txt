import discord
import asyncio
import os
from datetime import datetime

client = discord.Client()

if not os.path.exists('logs'):
    os.makedirs('logs')
if not os.path.exists('media'):
    os.makedirs('media')

message_cache = {}

@client.event
async def on_ready():
    print(f'Logged in as {client.user}')

@client.event
async def on_message(message):
    message_cache[message.id] = message

    if message.attachments:
        for attachment in message.attachments:
            file_name = f"media/{message.id}_{attachment.filename}"
            await attachment.save(file_name)

@client.event
async def on_message_delete(message):
    cached_message = message_cache.get(message.id)

    if cached_message:
        log_message = f"Deleted message:\n"
        log_message += f"Author: {cached_message.author}\n"
        log_message += f"Channel: {cached_message.channel}\n"
        log_message += f"Content: {cached_message.content}\n"
        log_message += f"Timestamp: {cached_message.created_at}\n"

        if cached_message.attachments:
            log_message += "Attachments:\n"
            for attachment in cached_message.attachments:
                log_message += f"- {attachment.filename}\n"

        with open(f"logs/deleted_messages_{datetime.now().strftime('%Y-%m-%d')}.txt", "a", encoding="utf-8") as log_file:
            log_file.write(log_message + "\n\n")

        del message_cache[message.id]

client.run('TOKEN')
